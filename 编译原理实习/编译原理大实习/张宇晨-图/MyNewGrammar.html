Start ::= ( "int" | "void" ) "main" "(" ")" "{" ( SentenceBlock )* "}"
SentenceBlock ::= Sentence
| Returnstatment
| "{" ( SentenceBlock )* "}"
Sentence ::= shunxu
| tiaojianyuju
| xunhuan
shunxu ::= ( ( shengming | ( fuzhi | zizeng ) ) ";" | SwitchCase )
shengming ::= ( "int" | "double" | "float" | "char" ) Identifier ( ( "=" biaodashi ) | ( "[" ( Integer | Identifier ) "]" ( "[" ( Integer | Identifier ) "]" )? ) )? ( "," Identifier ( ( "=" biaodashi ) | ( "[" ( Integer | Identifier ) "]" ( "[" ( Integer | Identifier ) "]" )? ) )? )*
fuzhi ::= Identifier ( "[" ( Integer | Identifier ) "]" ( "[" ( Integer | Identifier ) "]" )? "=" biaodashi )? ( ( "=" ( ( Identifier "[" ( Integer | Identifier ) "]" ( "[" ( Integer | Identifier ) "]" )? ) | ( biaodashi ) ) ) | ( "++" ) | ( "--" ) )?
zizeng ::= ( "++" Identifier )
| ( "--" Identifier )
tiaojianyuju ::= "if" "(" Boolean ")" SentenceBlock ( <ELSE> ( SentenceBlock ) )?
xunhuan ::= ( "while" "(" Boolean ")" SentenceBlock )
| ( <FOR> <LRBRACKET> ( ( ( "int" )? Identifier "=" biaodashi )? ";" ) Boolean ";" ( zizeng | fuzhi ) <RRBRACKET> ( ( "{" ( SentenceBlock )? "}" ) | ( ";" ) ) )
| ( <DO> "{" SentenceBlock "}" <WHILE> "(" Boolean ")" ";" ) 
/*
布尔表达式的实现
布尔表达式中有四种运算符，关系符、与、或、非
> < >= <= != == && || !
为了实现上述复杂的布尔表达式的翻译，定义了三个语法语义程序
BaiscBoolean():最简单的布尔表达式即包含非操作的条件表达式
YuBoolean():与&&结合起来的布尔表达式，由BasicBoolen()+&&组成
Boolean():或||结合起来的布尔表达式，由YuBoolean()+||组成
因为运算优先级 rop >! >&& >||
所以Boolean()即是最后可以匹配布尔表达式是语法
*/
BasicBoolean ::= ( ( <FEI> ( <LRBRACKET> tiaojian <RRBRACKET> ) ) | ( tiaojian ) )
YuBoolean ::= BasicBoolean ( <YU> YuBoolean )?
Boolean ::= YuBoolean ( <HUO> Boolean )?
tiaojian ::= biaodashi ( guanxifu biaodashi )? 
/*

{
  case value1:

  		break;
  case value2:
  		XXX;
  		break;
  default:
		XXX;
}

*/
SwitchCase ::= <SWITCH> "(" Expression ")" "{" ( <CASE> <INTEGER_LITERAL> ":" SentenceBlock )* ( <MINIDEFAULT> ":" SentenceBlock ) "}"
guanxifu ::= ( "<" | ">" | "=" | "<=" | ">=" | "==" | "!=" )

biaodashi ::= Expression
Expression ::= AdditiveExpression
AdditiveExpression ::= MultiplicativeExpression ( ( "+" | "-" ) MultiplicativeExpression )*
MultiplicativeExpression ::= UnaryExpression ( ( "*" | "/" | "%" ) UnaryExpression )*
UnaryExpression ::= "(" Expression ")"
| Identifier
| Integer
| Float
Identifier ::= <IDENTIFIER>
Integer ::= <INTEGER_LITERAL>
Returnstatment ::= <RETURN> biaodashi ";"
Float ::= <FLOAT_LITERAL>
